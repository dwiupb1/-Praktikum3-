# KODE PROGRAM MENGHITUNG LUAS DAN KELILING

r = float(input('Masukkan nilai jari-jari : '))

phi = 3.14
diameter = 2*r

luas = phi*r*r

keliling = phi*diameter

print('\nLuasnya    =', str("%.2f" % luas))

print('kelilingnya =', str("%.2f" % keliling))
